1 — Include library

<script src="https://tbljs.netlify.app/tbl.js"></script>

2 — Render from URL into a DIV

<div id="my-table"></div>
<script>
  TBL.loadUrl('https://yoursite.com/data.tbl',
              document.getElementById('my-table'));
</script>

3 — Render from <input type="file">

<input type="file" accept=".tbl" id="f">
<div id="out"></div>
<script>
  document.getElementById('f').onchange = function(){
    TBL.loadFile(this.files[0], document.getElementById('out'));
  };
</script>

4 — Parse binary manually

const buf = await fetch('data.tbl').then(r=>r.arrayBuffer());
const { columns, rows, meta } = TBL.parse(buf);
// columns: [{name,type,props}, ...]
// rows:    [[val,...], ...]
// meta:    { title, description, fontFamily, ... }

const objects = TBL.toObjects({ columns, rows });
// [{col1: val1, col2: val2}, ...]

5 — Encode to binary

const buf = TBL.encode({
  meta: { title: 'Products', fontFamily: 'Outfit' },
  columns: [
    { name: 'id',    type: TBL.TYPE.INT32     },
    { name: 'name',  type: TBL.TYPE.STRING    },
    { name: 'price', type: TBL.TYPE.FLOAT64   },
    { name: 'photo', type: TBL.TYPE.IMAGE_URL },
    { name: 'bg',    type: TBL.TYPE.COLOR     },
    { name: 'font',  type: TBL.TYPE.FONT      },
    { name: 'date',  type: TBL.TYPE.DATE      },
    { name: 'live',  type: TBL.TYPE.BOOL      },
  ],
  rows: [
    [1,'Laptop',999.99,'https://…/img.jpg','#1e40af','Syne','2024-03-01',true],
  ]
});

Column Types

TBL.TYPE.STRING    = 0   // UTF-8 text
TBL.TYPE.INT32     = 1   // 32-bit integer (4 bytes)
TBL.TYPE.FLOAT64   = 2   // 64-bit double  (8 bytes)
TBL.TYPE.BOOL      = 3   // boolean        (1 byte)
TBL.TYPE.IMAGE_URL = 4   // URL → rendered as <img>
TBL.TYPE.COLOR     = 5   // CSS color → color swatch
TBL.TYPE.DATE      = 6   // ISO date YYYY-MM-DD
TBL.TYPE.FONT      = 7   // Google Font name → loads & previews
