.TBL File Generator

Binary JSON table file creator

A .tbl file stores table data as binary-encoded JSON. This editor lets you create, export, import and preview TBL files.

tbl.js Library Code

class TBLTable{ constructor(data){ this.columns = data.columns || []; this.rows = data.rows || []; } toJSON(){ return { columns:this.columns, rows:this.rows }; } toBinary(){ const json = JSON.stringify(this.toJSON()); const encoder = new TextEncoder(); return encoder.encode(json); } download(filename="table.tbl"){ const blob = new Blob( [this.toBinary()], {type:"application/octet-stream"} ); const a = document.createElement("a"); a.href = URL.createObjectURL(blob); a.download = filename; a.click(); } } window.TBL = { Table:TBLTable, parse(buffer){ const decoder = new TextDecoder(); const json = decoder.decode(buffer); return JSON.parse(json); }, async fetch(url){ const res = await fetch(url); const buffer = await res.arrayBuffer(); const data = TBL.parse(buffer); return data; } };
